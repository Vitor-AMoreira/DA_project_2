139  double haversine(double latitude1, double longitude1, double latitude2, double longitude2) const
140  {
141  double latDiff = (latitude2 - latitude1) * toRadians;
142  double lonDiff = (longitude2 - longitude1) * toRadians;
143 
144  // Haversine formula
145  double a = sin(latDiff / 2) * sin(latDiff / 2) +
146  cos(latitude1) * cos(latitude2) *
147  sin(lonDiff / 2) * sin(lonDiff / 2);
148  double c = 2 * atan2(sqrt(a), sqrt(1 - a));
149  double distance = 6371.0 * c;
150 
151  return distance;
152  }
